import pandas as pd

def sentiment_score(text):
    text = str(text).lower()
    if "excellent" in text or "very satisfied" in text:
        return 0.95
    elif "great" in text or "helpful" in text:
        return 0.85
    elif "good" in text:
        return 0.65
    elif "issue" in text or "slow" in text:
        return 0.35
    elif "frustrating" in text or "unclear" in text:
        return 0.2
